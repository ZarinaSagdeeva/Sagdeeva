from task_3_box import Box

class Truck:
    def __init__(self, capacity):
        self.__capacity = capacity
    
    def __str__(self):
        return f"Truck with capacity {len(self.__capacity)} boxes"
    
    def add(self, box):
        self.__capacity.append(box)
    
    def sub(self, box):
        self.__capacity.remove(box)

box1 = Box("John Smith", "New York", "Los Angeles")
box2 = Box("Jane Doe", "San Francisco", "Seattle")
box3 = Box("Bob Johnson", "Chicago", "Miami")

print(box1.get_name()) # Output: John Smith
print(box2.get_target_city()) # Output: Seattle
print(box3.get_postcode()) # Output: random number between 10000 and 99999

truck = Truck([box1, box2])
print(truck) # Output: Truck with capacity 2 boxes

truck.add(box3)
print(truck) # Output: Truck with capacity 3 boxes

truck.sub(box1)
print(truck) # Output: Truck with capacity 2 boxes
