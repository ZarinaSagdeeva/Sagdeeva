import random

class Box:
    def __init__(self, name, from_city, target_city):
        self.__postcode = random.randint(10000, 99999)
        self.__name = name
        self.__from_city = from_city
        self.__target_city = target_city
    
    def get_postcode(self):
        return self.__postcode
    
    def get_name(self):
        return self.__name
    
    def get_from_city(self):
        return self.__from_city
    
    def get_target_city(self):
        return self.__target_city
    
    def set_target_city(self, target_city):
        self.__target_city = target_city

box1 = Box("John Smith", "New York", "Los Angeles")

print(box1.get_postcode()) # Output: random number between 10000 and 99999
print(box1.get_name()) # Output: John Smith
print(box1.get_from_city()) # Output: New York
print(box1.get_target_city()) # Output: Los Angeles

box1.set_target_city("San Francisco")

print(box1.get_target_city()) # Output: San Francisco
